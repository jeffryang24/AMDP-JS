Array and Object

                    
// Initialize Array

var arr2 = [];  // recommended way
var arr3 = new Array('alpha', 'beta', 'charlie');
var arr4 = ['delta', 'echo', 'foxtrot'];

// Indexed Array

// Associative Array
var arr5 = { nama: 'Tango Oreo', kelas: 'VII' };
document.write(arr5['nama']);
document.write(arr5.nama);

// Complex Nested Array
var arr6 = {
    nama: "John Titor",
    sekolah: [{
        nama: 'SMP Teo',
        avg: 89
    }, {
        nama: 'SMA Teo',
        avg: 90
    }],
    nickname: ['Bocah', 'Entong', 'Joe']
};

document.write(arr6.sekolah[0].nama);

Loop Condition

                    
// For Loop
for(let i = 0; i < 100; i++) {
    document.write(i);
}

// Do While Loop
var a = 0;
do {
    document.write(new Date().toUTCString());
    a++;
}while(a < 10)

// While Do Loop
a = 10
while(a > 0) {
    document.write('test');
    a--;
}

// Accessing array using Loop
var arr1 = ['Saya', 'Suka', 'Main', 'Catur'];
for(let i = 0; i < arr1.length; i++) {
    document.write(arr1[i]);
}
